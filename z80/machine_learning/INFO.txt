"CALC INFO"
ClrHome
Input "NUM OF LABELS?",C

SetUpEditor ʟINFO
C→dim(ʟINFO)

"FIND HOW MUCH IN EACH LABEL"
For(I,1,C,1)
  ClrHome
  Output(1,1,"LABEL")
  Output(2,1,I)
  Pause 
  
  ClrHome
  Input "NUM IN LABEL?",X
  X→ʟINFO(I)
End

"CALC TOTAL ITEMS"
sum(ʟINFO)→T

0→Z
For(I,1,C,1)
  If ʟINFO(I)≠0:Then
  Z-(ʟINFO(I)/T)*log(ʟINFO(I)/T,2)→Z
  End
End

Disp "INFO",Z

"CLEAN UP"
DelVar ʟINFO
